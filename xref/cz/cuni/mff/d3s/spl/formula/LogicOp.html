17  package cz.cuni.mff.d3s.spl.formula;
18  
19  import cz.cuni.mff.d3s.spl.Result;
20  import cz.cuni.mff.d3s.spl.data.DataSource;
21  import cz.cuni.mff.d3s.spl.interpretation.Interpretation;
22  
23  import java.util.HashSet;
24  import java.util.NoSuchElementException;
25  import java.util.Set;
26  
27  /** Formula node: common parent for all logic operations.
28   *
29   */
30  abstract class LogicOp implements Formula {
31  	protected Formula left;
32  	protected Formula right;
33  	protected Set<String> variables;
34  	
35  	public LogicOp(Formula left, Formula right) {
36  		this.left = left;
37  		this.right = right;
38  		variables = new HashSet<>();
39  		variables.addAll(left.getVariables());
40  		variables.addAll(right.getVariables());
41  	}
42  
43  	@Override
44  	public void setInterpretation(Interpretation interpretation) {
45  		left.setInterpretation(interpretation);
46  		right.setInterpretation(interpretation);
47  	}
48  
49  	@Override
50  	public void bind(String variable, DataSource data) {
51  		bindToMultiple(variable, data, left, right);
52  	}
53  	
54  	static final void bindToMultiple(String variable, DataSource data, Formula... formulas) {
55  		boolean somewhereBinded = false;
56  		RuntimeException lastException = null;
57  		
58  		for (Formula f : formulas) {
59  			try {
60  				f.bind(variable, data);
61  				somewhereBinded = true;
62  			} catch (NoSuchElementException e) {
63  				lastException = e;
64  			}
65  		}
66  		if (!somewhereBinded) {
67  			if (lastException != null) {
68  				throw lastException;
69  			}
70  		}
71  	}
72  	
73  	@Override
74  	abstract public Result evaluate(double significanceLevel);
75  
76  	@Override
77  	public Set<String> getVariables() {
78  		return variables;
79  	}
80  }
