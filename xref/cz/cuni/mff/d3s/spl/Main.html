34  public class Main {
35  	private static final double SIGNIFICANCE_LEVEL = 0.95;
36  
37  	public static void main(String[] args) {
38  		CommandLineParser parser = new DefaultParser();
39  		Options options = createOptions();
40  
41  		try {
42  			CommandLine line = parser.parse(options, args);
43  
44  			String jarFormulas = line.getOptionValue("jar-formulas");
45  			String fileFormulas = line.getOptionValue("file-formulas");
46  			String[] cmdFormulas = line.getOptionValues("commandline-formulas");
47  			String dataDir = line.getOptionValue("data-dir");
48  			String revisionMapping = line.getOptionValue("revision-mapping");
49  			boolean printUnknownOnly = line.hasOption("print-unknown");
50  			boolean startServer = line.hasOption("server");
51  
52  
53  			// Formulas are processed in order jar, file and command line.
54  			// Latter options have higher priority and will override previous values.
55  			// Pair of FQ benchmark name and corresponding SPL formula.
56  			Map<String, String> formulas = new HashMap<>();
57  			readJarFormulas(formulas, jarFormulas);
58  			readFileFormulas(formulas, fileFormulas);
59  			readCommandlineFormulas(formulas, cmdFormulas);
60  
61  			DataReader reader = new StructuredDataReader<>(new JmhJsonRevisionReader.RevisionFactory());
62  			Map<DataInfo, List<Revision>> data = reader.readData(new String[]{dataDir});
63  
64  			if (startServer) {
65  				// Start REST API server to allow fetching data from visualization tools
66  				TestsApiServiceFactory.setService(new TestsApiServiceImpl(data));
67  				new MicroservicesRunner(42000)
68  						.addInterceptor(new CorsInterceptor())
69  						.deploy(new TestsApi())
70  						.deploy(new WebVisualizer())
71  						.start();
72  				if (Desktop.isDesktopSupported()) {
73  					Desktop.getDesktop().browse(new URI("http://localhost:42000/index.html"));
74  				}
75  			} else {
76  				// Get custom mapping of revisions form file.
77  				Map<String, String> customRevisionMap = getCustomRevisionMapping(revisionMapping);
78  
79  				for (Map.Entry<DataInfo, List<Revision>> benchmarkItem : data.entrySet()) {
80  					String benchmarkName = benchmarkItem.getKey().getId();
81  					String formulaString = null;
82  
83  					// Wildcard method identifier for all methods
84  					if (formulas.containsKey("*")) {
85  						formulaString = formulas.get("*");
86  					}
87  
88  					// Method fully qualified name without additional params
89  					String benchmarkBasename = benchmarkName.substring(0, benchmarkName.indexOf('@'));
90  					if (formulas.containsKey(benchmarkBasename)) {
91  						formulaString = formulas.get(benchmarkBasename);
92  					}
93  
94  					// Method fully qualified name including additional info like measurement mode or execution params
95  					if (formulas.containsKey(benchmarkName)) {
96  						formulaString = formulas.get(benchmarkName);
97  					}
98  
99  					if (formulaString == null) {
100 						System.out.println("Skipping benchmark without formula: " + benchmarkItem.getKey());
101 						continue;
102 					}
103 
104 					Formula formula = SplFormula.create(formulaString);
105 					formula.setInterpretation(new WelchTestInterpretation());
106 
107 					// get benchmark's revisions in better format for us
108 					Map<String, DataSource> revisionMap = DataUtils.getRevisionMap(benchmarkItem.getValue());
109 					boolean isUnknown = false;
110 					// Bind variables to formula (according to revisions collection)
111 					for (String variable : formula.getVariables()) {
112 						// try if there is real revision of that name
113 						if (revisionMap.containsKey(variable)) {
114 							formula.bind(variable, revisionMap.get(variable));
115 						}
116 						// else try to find another revision is custom mapping
117 						else if (customRevisionMap.containsKey(variable)) {
118 							formula.bind(variable, revisionMap.get(customRevisionMap.get(variable)));
119 						}
120 						// else there is no such revision
121 						else {
122 							System.out.printf("Benchmark: %s, formula: %s, unknown version: %s\n", benchmarkName,
123 									formulaString, variable);
124 							isUnknown = true;
125 						}
126 					}
127 
128 					// if we want only get missing versions, skip formula evaluating
129 					// (or if there are some missing ones, evaluation make no sense)
130 					if (!printUnknownOnly && !isUnknown) {
131 						Result result = formula.evaluate(SIGNIFICANCE_LEVEL);
132 						System.out.printf("Benchmark: %s, formula: %s, result: %s\n", benchmarkName, formulaString, result);
133 					}
134 				}
135 			}
136 		} catch (ParseException e) {
137 			// oops, something went wrong
138 			System.err.println("Parsing failed. Reason: " + e.getMessage());
139 			printHelp(options);
140 		} catch (DataReader.ReaderException e) {
141 			System.err.println("Cannot read measured data. Reason: " + e.getMessage());
142 		} catch (IOException | URISyntaxException e) {
143 			System.err.println("Cannot open your default browser. Reason: " + e.getMessage());
144 		}
145 	}

147 	private static void readJarFormulas(Map<String, String> formulas, String jarFormulas) {
148 		if (jarFormulas != null) {
149 			try {
150 				File jarFile = new File(jarFormulas);
151 				URL url = new URL("jar:file:" + jarFile.getAbsolutePath() + "!/META-INF/SPLFormulas");
152 				InputStream is = url.openStream();
153 				BufferedReader reader = new BufferedReader(new InputStreamReader(is));
154 				String line;
155 				List<String> lines = new ArrayList<>();
156 				while ((line = reader.readLine()) != null) {
157 					lines.add(line);
158 				}
159 				parseStringsIntoMap(lines, formulas);
160 			} catch (IOException | NullPointerException e) {
161 				System.err.println("Cannot read SPL formulas from JAR, assuming no formulas.");
162 			}
163 		}
164 	}
165 
166 	private static void readFileFormulas(Map<String, String> formulas, String fileFormulas) {
167 		try {
168 			List<String> lines = Files.readAllLines(Paths.get(fileFormulas), StandardCharsets.UTF_8);
169 			parseStringsIntoMap(lines, formulas);
170 		} catch (IOException e) {
171 			System.err.println("Cannot read SPL formulas from file, assuming empty one.");
172 		} catch (NullPointerException e) {
173 			// there is no such formula file
174 		}
175 	}
176 
177 	private static void readCommandlineFormulas(Map<String, String> formulas, String[] cmdFormulas) {
178 		if (cmdFormulas != null) {
179 			parseStringsIntoMap(cmdFormulas, formulas);
180 		}
181 	}

183 	private static void parseStringsIntoMap(String[] data, Map<String, String> output) {
184 		for (String line : data) {
185 			// skip empty lines and comments
186 			if (line.equals("") || line.startsWith("#")) {
187 				continue;
188 			}
189 			String[] parsed = line.split(":");
190 			if (parsed.length == 2) {
191 				// save only well formed lines
192 				output.put(parsed[0], parsed[1]);
193 			} else {
194 				System.err.println("Malformed line '" + line + "'. Exactly one ':' expected.");
195 			}
196 		}
197 	}
198 
199 	private static void parseStringsIntoMap(Collection<String> data, Map<String, String> output) {
200 		parseStringsIntoMap(data.toArray(new String[] {}), output);
201 	}
202 
203 	private static Map<String, String> getCustomRevisionMapping(String revisionsMappingFile) {
204 		List<String> lines;
205 		try {
206 			lines = Files.readAllLines(Paths.get(revisionsMappingFile), StandardCharsets.UTF_8);
207 			Map<String, String> result = new HashMap<>();
208 			parseStringsIntoMap(lines, result);
209 			return result;
210 		} catch (IOException e) {
211 			System.err.println("Cannot read mapping file, assuming empty one.");
212 		} catch (NullPointerException e) {
213 			// there is no such mapping file, return empty mapping
214 		}
215 		return new HashMap<>();
216 	}

225 	private static Options createOptions() {
226 		Options options = new Options();
227 
228 		options.addOption(Option.builder("j")
229 				.longOpt("jar-formulas")
230 				.hasArg()
231 				.argName("benchmarks.jar")
232 				.desc("Read SPL formulas from JAR file.")
233 				.build()
234 		);
235 
236 		options.addOption(Option.builder("f")
237 				.longOpt("file-formulas")
238 				.hasArg()
239 				.argName("formula_file")
240 				.desc("Read SPL formulas from text file.")
241 				.build()
242 		);
243 
244 		options.addOption(Option.builder("c")
245 				.longOpt("commandline-formulas")
246 				.hasArgs()
247 				.argName("formulas")
248 				.desc("Read SPL formulas from commandline arguments.")
249 				.build()
250 		);
251 
252 		options.addOption(Option.builder("d")
253 				.longOpt("data-dir")
254 				.hasArg()
255 				.argName("data_directory")
256 				.required()
257 				.desc("Path to directory with measured data.")
258 				.build()
259 		);
260 
261 		options.addOption(Option.builder("r")
262 				.longOpt("revision-mapping")
263 				.hasArg()
264 				.argName("mapping_file")
265 				.desc("Mapping of formula revisions to file names.")
266 				.build()
267 		);
268 
269 		options.addOption(Option.builder("p")
270 				.longOpt("print-unknown")
271 				.desc("Print unknown revisions and exit.")
272 				.build()
273 		);
274 
275 		options.addOption(Option.builder("S")
276 				.longOpt("server")
277 				.desc("Start local server for data visualization.")
278 				.build()
279 		);
280 
281 		return options;
282 	}
