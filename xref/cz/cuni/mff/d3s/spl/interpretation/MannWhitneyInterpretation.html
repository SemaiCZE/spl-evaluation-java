17  package cz.cuni.mff.d3s.spl.interpretation;
18  
19  import cz.cuni.mff.d3s.spl.data.BenchmarkRun;
20  import cz.cuni.mff.d3s.spl.data.BenchmarkRunUtils;
21  import cz.cuni.mff.d3s.spl.data.DataSnapshot;
22  import org.apache.commons.math3.distribution.NormalDistribution;
23  import org.apache.commons.math3.stat.inference.MannWhitneyUTest;
24  
25  /** SPL interpretation based on Mann-Whitney test.
26   * 
27   * This code is greatly inspired by
28   * org.apache.commons.math3.stat.inference.MannWhitneyUTest implementation
29   * that could not be used directly in the compare() method.
30   *
31   */
32  public class MannWhitneyInterpretation implements Interpretation {
33  	private final MannWhitneyUTest utest = new MannWhitneyUTest();
34  	
35  	/** {@inheritDoc} */
36  	@Override
37  	public ComparisonResult compare(DataSnapshot left, DataSnapshot right) {
38  		double[] leftSamples = mergeSamples(left);
39  		double[] rightSamples = mergeSamples(right);
40  		
41  		double uStatMax = utest.mannWhitneyU(leftSamples, rightSamples);
42  		
43  		long lengthsMultiplied = (long) leftSamples.length * rightSamples.length;
44  
45  		double uStatMin = lengthsMultiplied - uStatMax;
46  		
47  		/* https://en.wikipedia.org/wiki/Mann%E2%80%93Whitney_U_test#Normal_approximation */
48  		double meanU = lengthsMultiplied / 2.0;
49  		double varU = lengthsMultiplied * (leftSamples.length + rightSamples.length + 1) / 12.0;
50  		
51  		double z = (uStatMin - meanU) / Math.sqrt(varU);
52  		
53  		NormalDistribution distribution = new NormalDistribution(0.0, 1.0);
54  		
55  		return new DistributionBasedComparisonResult(z, distribution);
56  	}
57  
58  	/** {@inheritDoc} */
59  	@Override
60  	public ComparisonResult compare(DataSnapshot data, double value) {
61  		throw new UnsupportedOperationException("This is not yet implemented.");
62  	}
63  
64  	private double[] mergeSamples(DataSnapshot data) {
65  		BenchmarkRun merged = BenchmarkRunUtils.merge(data.getRuns());
66  		
67  		return BenchmarkRunUtils.toDoubleArray(merged);
68  	}
69  }
