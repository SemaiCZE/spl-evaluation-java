17  package cz.cuni.mff.d3s.spl.interpretation;
18  
19  /** Encapsulates result of a comparison of two data sources.
20   * 
21   * <p>
22   * The basic operation is to get what is the relation between the
23   * two data sets.
24   * That is the only operation that must be provided by the interface,
25   * all other methods may throw UnsupportedOperationException.
26   * 
27   * <p>
28   * For a typical statistical test the comparison result is directly
29   * derived from the statistic and its position relative to the critical
30   * region as computed at given significance level.
31   *
32   */
33  public interface ComparisonResult {
34  	/** Actual result of the comparison.
35  	 * 
36  	 * All the values are bound to the given significance level.
37  	 */
38  	public enum Relation {
39  		/** It is not possible to tell the relation at given significance level. */
40  		UNKNOWN {
41  			@Override
42  			public String toShortString() {
43  				return "?";
44  			}
45  		},
46  		/** The first (left) data set is smaller than the second (right) one. */
47  		LESS_THAN {
48  			@Override
49  			public String toShortString() {
50  				return "<";
51  			}
52  		},
53  		/** The data sets are considered equal. */
54  		EQUAL {
55  			@Override
56  			public String toShortString() {
57  				return "=";
58  			}
59  		},
60  		/** The first (left) data set is greater than the second (right) one. */
61  		GREATER_THAN {
62  			@Override
63  			public String toShortString() {
64  				return ">";
65  			}
66  		};
67  		
68  		/** Get relation as short string.
69  		 *
70  		 * @return Relation as short string.
71  		 */
72  		public abstract String toShortString();
73  	};
74  	
75  	/** Get the actual result.
76  	 * 
77  	 * @param significanceLevel Significance level to use.
78  	 * @return Relation between the two original data sets.
79  	 */
80  	public Relation get(double significanceLevel);
81  	
82  	/** Get statistic of the underlying test.
83  	 * 
84  	 * @return Test statistic.
85  	 * @throws UnsupportedOperationException When the operation is not supported.
86  	 */
87  	public double getStatistic();
88  	
89  	/** Tells a critical value for given significance level.
90  	 * 
91  	 * @param significanceLevel Significance level to use.
92  	 * @return Critical value at given significance level.
93  	 * @throws UnsupportedOperationException When the operation is not supported.
94  	 * @throws IllegalArgumentException When the significanceLevel is outside range 0 and 1.
95  	 */
96  	public double getCriticalValue(double significanceLevel);
97  	
98  	/** Tells confidence interval for given level of the interpretation.
99  	 * 
100 	 * <p>
101 	 * Typically the confidence interval is returned for a difference
102 	 * between means of the two data sets.
103 	 * 
104 	 * @param confidenceLevel Confidence level to use.
105 	 * @return Typically a two member array denoting the confidence interval.
106 	 * @throws UnsupportedOperationException When the operation is not supported.
107 	 * @throws IllegalArgumentException When the significanceLevel is outside range 0 and 1.
108 	 */
109 	public double[] getConfidenceInterval(double confidenceLevel);
110 }
