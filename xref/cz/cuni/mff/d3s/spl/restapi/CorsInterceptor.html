1   package cz.cuni.mff.d3s.spl.restapi;
2   
3   import org.wso2.msf4j.Interceptor;
4   import org.wso2.msf4j.Request;
5   import org.wso2.msf4j.Response;
6   import org.wso2.msf4j.ServiceMethodInfo;
7   
8   public class CorsInterceptor implements Interceptor {
9   
10  	@Override
11  	public boolean preCall(Request request, Response response, ServiceMethodInfo serviceMethodInfo) throws Exception {
12  		response.setHeader("Access-Control-Allow-Origin", "*");
13  		response.setHeader("Access-Control-Allow-Credentials", "true");
14  		response.setHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS, DELETE, PUT");
15  		response.setHeader("Access-Control-Max-Age", "3600");
16  		response.setHeader("Access-Control-Allow-Headers", "Content-Type, Accept, X-Requested-With, remember-me");
17  
18  		if (request.getHeader("Origin") != null && !request.getHeader("Origin").isEmpty()) {
19  			response.setHeader("Access-Control-Allow-Origin", request.getHeader("Origin"));
20  		}
21  		return true;
22  	}
23  
24  	@Override
25  	public void postCall(Request request, int status, ServiceMethodInfo smi) throws Exception {
26  	}
27  
28  }
