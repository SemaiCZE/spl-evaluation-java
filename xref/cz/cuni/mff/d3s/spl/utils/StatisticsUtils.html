17  package cz.cuni.mff.d3s.spl.utils;
18  
19  import org.apache.commons.math3.stat.descriptive.moment.Mean;
20  import org.apache.commons.math3.stat.descriptive.moment.Variance;
21  
22  import java.util.Random;
23  
24  /** Helper methods for common statistics.
25   */
26  public class StatisticsUtils  {
27  	/** Compute arithmetic mean of given data.
28  	 * 
29  	 * @param values Array of values to compute the mean from.
30  	 * @return Mean of the provided values.
31  	 */
32  	public static double mean(double... values) {
33  		Mean mean = new Mean();
34  		return mean.evaluate(values);
35  	}
36  	
37  	/** Compute variance of given data with bias correction.
38  	 * 
39  	 * @param values Array of values to compute the variance from.
40  	 * @return Varince of the provided values.
41  	 */
42  	public static double variance(double... values) {
43  		Variance var = new Variance();
44  		return var.evaluate(values);
45  	}
46  	
47  	/** Compute variance of given data without bias correction.
48  	 * 
49  	 * @param values Array of values to compute the variance from.
50  	 * @return Varince of the provided values.
51  	 */
52  	public static double varianceN(double... values) {
53  		Variance var = new Variance(false);
54  		return var.evaluate(values);
55  	}
56  	
57  	/** Bootstrap from already known values.
58  	 * 
59  	 * <p>
60  	 * The bootstrapping procedure is simple as we randomly select values from
61  	 * the source array to store in the destination one.
62  	 * Obviously, the individual values can be repeated in the bootstrapped
63  	 * array.
64  	 * 
65  	 * @param source Array with original values to bootstrap from.
66  	 * @param dest Array where to store the bootstrapped values.
67  	 * @param rnd Random number generator to use.
68  	 */
69  	public static void bootstrap(double[] source, double dest[], Random rnd) {
70  		for (int i = 0; i < dest.length; i++) {
71  			dest[i] = source[rnd.nextInt(source.length)];
72  		}
73  	}
74  }
