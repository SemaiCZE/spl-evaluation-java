10  /**
11   * Provides unified interface to read data revision from files.
12   *
13   * Each implementation will specify it's semantics in detail,
14   * but common idea is that each reader processes data from
15   * given file(s) and returns them as a map with method/benchmark
16   * name as a key and data as a value.
17   */
18  public interface RevisionReader {
19  
20  	/**
21  	 * Read one revision of data from given files. There shouldn't be
22  	 * more revisions of the same benchmark/method/... data in one file
23  	 * or you shouldn't pass multiple files containing the same data with
24  	 * different revisions.
25  	 *
26  	 * @param files Input files with raw data
27  	 * @return Processed data as a map grouped by benchmark/method/...
28  	 *          If there is no info about name in the data, "default"
29  	 *          identifier is used instead.
30  	 */
31  	Map<DataInfo, DataSource> readRevision(File... files) throws DataReader.ReaderException;
32  }
