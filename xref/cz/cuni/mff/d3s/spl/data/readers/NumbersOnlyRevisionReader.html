15  /**
16   * Reader for text files with one integer number per line.
17   *
18   * This reader expects multiple files (each is separate benchmark run)
19   * with long numbers in binary form. There're no data about naming,
20   * so it's expected that all data are from the same benchmark
21   * and results are returned with the key "default".
22   */
23  public class NumbersOnlyRevisionReader implements RevisionReader {
24  	/**
25  	 * Read one revision of data from given files. It's expected that
26  	 * each file is separate run and all data are from the same benchmark.
27  	 *
28  	 * @param files Input files with raw data
29  	 * @return Processed data with key "default"
30  	 * @throws ReaderException On reading or parsing error
31  	 */
32  	@Override
33  	public Map<DataInfo, DataSource> readRevision(File... files) throws ReaderException {
34  		DataSnapshotBuilder snapshotBuilder = new DataSnapshotBuilder();
35  		for (File file : files) {
36  			try {
37  				BenchmarkRun run = readNumbersOnlyData(new FileInputStream(file)).create();
38  				snapshotBuilder.addRun(run);
39  			} catch (FileNotFoundException e) {
40  				throw new ReaderException("File not found: " + e.getMessage());
41  			}
42  		}
43  
44  		HashMap<DataInfo, DataSource> result = new HashMap<>();
45  		result.put(DataInfo.defaultInstance, new BuilderDataSource(snapshotBuilder));
46  		return result;
47  	}
48  
49  	public static class RevisionFactory implements Factory<NumbersOnlyRevisionReader> {
50  		@Override
51  		public NumbersOnlyRevisionReader getInstance() {
52  			return new NumbersOnlyRevisionReader();
53  		}
54  	}
55  
56  	/** Reads samples from a data stream.
57  	 *
58  	 * Expects samples as numbers of long type in input stream. Reading ends when
59  	 * there's no next number in the stream.
60  	 *
61  	 * @param is Input stream with data.
62  	 * @return Benchmark run with the samples.
63  	 * @throws ReaderException on reading failure.
64  	 */
65  	private BenchmarkRunBuilder readNumbersOnlyData(InputStream is) throws ReaderException {
66  		BenchmarkRunBuilder run = new BenchmarkRunBuilder();
67  
68  		Scanner sc = new Scanner(is);
69  		while (sc.hasNextLong()) {
70  			run.addSamples(sc.nextLong());
71  		}
72  		sc.close();
73  
74  		return run;
75  	}
76  }
