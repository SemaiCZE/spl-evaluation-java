11  /**
12   * Reader for text files with one integer number per line.
13   *
14   * This reader expects multiple files (each is separate benchmark run)
15   * with one integer (long type) number per line. There're no data about
16   * naming, so it's expected that all data are from the same benchmark
17   * and results are returned with the key "default".
18   */
19  public class LineOrientedRevisionReader implements RevisionReader {
20  	/**
21  	 * Read one revision of data from given files. It's expected that
22  	 * each file is separate run and all data are from the same benchmark.
23  	 *
24  	 * @param files Input files with raw data
25  	 * @return Processed data with key "default"
26  	 * @throws ReaderException On reading or parsing error
27  	 */
28  	@Override
29  	public Map<DataInfo, DataSource> readRevision(File... files) throws ReaderException {
30  		DataSnapshotBuilder snapshotBuilder = new DataSnapshotBuilder();
31  		for (File file : files) {
32  	    	try {
33  	    		BenchmarkRun run = readLineOrientedData(new FileInputStream(file)).create();
34  	    		snapshotBuilder.addRun(run);
35  	    	} catch (FileNotFoundException e) {
36  			    throw new ReaderException("File not found: " + e.getMessage());
37  	    	}
38  		}
39  
40  		HashMap<DataInfo, DataSource> result = new HashMap<>();
41  		result.put(DataInfo.defaultInstance, new BuilderDataSource(snapshotBuilder));
42  		return result;
43  	}
44  
45  	public static class RevisionFactory implements Factory<LineOrientedRevisionReader> {
46  		@Override
47  		public LineOrientedRevisionReader getInstance() {
48  			return new LineOrientedRevisionReader();
49  		}
50  	}
51  
52  	/** Reads samples from a data stream.
53  	 *
54  	 * Expects each sample is on a separate line, silently ignores
55  	 * lines containing something else than positive integer.
56  	 *
57  	 * @param is Input stream with data.
58  	 * @return Benchmark run with the samples.
59  	 * @throws ReaderException on reading failure.
60  	 */
61  	private BenchmarkRunBuilder readLineOrientedData(InputStream is) throws ReaderException {
62  		BufferedReader reader = new BufferedReader(new InputStreamReader(is));
63  		BenchmarkRunBuilder run = new BenchmarkRunBuilder();
64  		String line;
65  
66  		try {
67  			while ((line = reader.readLine()) != null) {
68  				// skip empty lines and comments
69  				if (!line.equals("") && !line.startsWith("#")) {
70  					long value = Long.parseLong(line);
71  					run.addSamples(value);
72  				}
73  			}
74  		} catch (NumberFormatException e) {
75  			throw new ReaderException("Wrong number format: " + e.getMessage());
76  		} catch (IOException e) {
77  			throw new ReaderException("IO error: " + e.getMessage());
78  		}
79  
80  		return run;
81  	}
