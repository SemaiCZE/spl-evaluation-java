14  /**
15   * Data reader from plain text formats. See readData() documentation for
16   * more info about requested data format, hierarchical structure and revision
17   * reader requirements.
18   */
19  public class PlainDataReader<T extends RevisionReader> implements DataReader {
20  	/**
21  	 * Revision reader instance.
22  	 */
23  	private T reader;
24  
25  	/**
26  	 * Constructor which creates revision reader instance.
27  	 *
28  	 * @param readerFactory Factory for creating instance of revision reader.
29  	 *                      The instance must be of the same type as generic
30  	 *                      type T.
31  	 */
32  	public PlainDataReader(Factory<T> readerFactory) {
33  		reader = readerFactory.getInstance();
34  	}
35  
36  	/**
37  	 * Reads multiple revision data from files. Each argument is directory containing
38  	 * multiple files, one file per benchmark run. Runs in each directory are joined
39  	 * to a single revision. Revision order preserves input arguments order. Revision
40  	 * name is set as corresponding directory name. RevisionReader of T type is expected
41  	 * to return map with "default" key and single DataSource value.
42  	 *
43  	 * @param args Array of 1 string with path to root directory of measured data.
44  	 * @return Map with name of measured unit as key and list of revisions for that
45  	 *          unit as value.
46  	 */
47  	@Override
48  	public Map<DataInfo, List<Revision>> readData(String[] args) throws ReaderException {
49  		Map<DataInfo, List<Revision>> data = new HashMap<>();
50  		data.put(DataInfo.defaultInstance, new LinkedList<>());
51  
52  		for (String dirname : args) {
53  			File dir = new File(dirname);
54  
55  			System.out.printf("Reading data from %s...", dirname);
56  			Map<DataInfo, DataSource> revision = reader.readRevision(dir.listFiles());
57  			System.out.printf(" ok, %d run(s).\n", revision.get(DataInfo.defaultInstance).makeSnapshot().getRunCount());
58  
59  			data.get(DataInfo.defaultInstance).add(new Revision(dir.getName(), revision.get(DataInfo.defaultInstance)));
60  		}
61  
62  		return data;
63  	}
64  }
