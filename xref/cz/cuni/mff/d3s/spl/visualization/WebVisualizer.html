3   import org.apache.commons.io.IOUtils;
4   
5   import javax.ws.rs.core.Response;
6   import javax.ws.rs.*;
7   import java.io.IOException;
8   import java.io.InputStream;
9   
10  
11  @Path("/")
12  public class WebVisualizer {
13  
14  	@GET
15  	@Path("/{file : [a-zA-Z_0-9.-]*}")
16  	public Response explicitFiles(@PathParam("file") String requestedFile) {
17  		return serveFile(requestedFile);
18  	}
19  
20  	@GET
21  	@Path("/{dir : [a-zA-Z_0-9]*}/{file : [a-zA-Z_0-9.-]*}")
22  	public Response explicitFiles(@PathParam("dir") String dir, @PathParam("file") String requestedFile) {
23  		return serveFile(dir + "/" + requestedFile);
24  	}
25  
26  	@GET
27  	@Path("/{dir1 : [a-zA-Z_0-9]*}/{dir2 : [a-zA-Z_0-9]*}/{file : [a-zA-Z_0-9.-]*}")
28  	public Response explicitFiles(@PathParam("dir1") String dir1, @PathParam("dir2") String dir2, @PathParam("file") String requestedFile) {
29  		return serveFile(dir1 + "/" + dir2 + "/" + requestedFile);
30  	}
31  
32  	@GET
33  	public Response defaultFile() {
34  		return serveFile("index.html");
35  	}
36  
37  	private Response serveFile(String fileName) {
38  		try {
39  			InputStream content = getClass().getResourceAsStream("/cz/cuni/mff/d3s/spl/visualization/" + fileName);
40  			return Response.ok().entity(content).build();
41  		} catch (NullPointerException e) {
42  			return Response.status(404).entity("<h1>Not Found</h1>").build();
43  		}
44  	}
45  }
