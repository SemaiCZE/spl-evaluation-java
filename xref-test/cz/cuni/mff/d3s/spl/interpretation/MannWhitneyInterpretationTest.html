17  package cz.cuni.mff.d3s.spl.interpretation;
18  
19  import cz.cuni.mff.d3s.spl.data.BenchmarkRun;
20  import cz.cuni.mff.d3s.spl.data.DataSnapshot;
21  import cz.cuni.mff.d3s.spl.data.DataSnapshotBuilder;
22  import cz.cuni.mff.d3s.spl.data.ImmutableBenchmarkRun;
23  import org.junit.Before;
24  import org.junit.Test;
25  
26  import static org.junit.Assert.assertEquals;
27  
28  public class MannWhitneyInterpretationTest  {
29  	private DataSnapshot SNAPSHOT_1;
30  	private DataSnapshot SNAPSHOT_1_PERM;
31  	private DataSnapshot SNAPSHOT_2;
32  	private MannWhitneyInterpretation interpretation;
33  	
34  	private static final double[] RUN_1 = new double[] {1, 2, 3, 0 };
35  	private static final double[] RUN_1_PERM = new double[] {2, 1, 0, 3 };
36  	private static final double[] RUN_2 = new double[] {7, 5, 6, 8 };
37  	
38  	@Before
39  	public void prepareSnapshots() {
40  		SNAPSHOT_1 = makeSnapshotFromSingleRun(RUN_1);
41  		SNAPSHOT_1_PERM = makeSnapshotFromSingleRun(RUN_1_PERM);
42  		SNAPSHOT_2 = makeSnapshotFromSingleRun(RUN_2);
43  	}
44  	
45  	@Before
46  	public void prepareInterpretation() {
47  		interpretation = new MannWhitneyInterpretation();
48  	}
49  	
50  	private DataSnapshot makeSnapshotFromSingleRun(double... samples) {
51  		BenchmarkRun run = new ImmutableBenchmarkRun(samples);
52  		
53  		DataSnapshotBuilder builder = new DataSnapshotBuilder();
54  		builder.addRun(run);
55  		
56  		return builder.create();
57  	}
58  		
59  	
60  	@Test
61  	public void smokeTestForTwoSnapshots() {
62  		ComparisonResult result = interpretation.compare(SNAPSHOT_1, SNAPSHOT_2);
63  		
64  		assertEquals(-2.309, result.getStatistic(), 0.001);
65  		assertEquals(ComparisonResult.Relation.LESS_THAN, result.get(0.9));
66  	}
67  	
68  	@Test
69  	public void equalSamples() {
70  		ComparisonResult result = interpretation.compare(SNAPSHOT_1, SNAPSHOT_1_PERM);
71  		
72  		assertEquals(0.0, result.getStatistic(), 0.001);
73  		assertEquals(ComparisonResult.Relation.EQUAL, result.get(0.9));
74  	}
75  }
