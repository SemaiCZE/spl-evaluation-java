17  package cz.cuni.mff.d3s.spl.interpretation;
18  
19  import cz.cuni.mff.d3s.spl.data.BenchmarkRun;
20  import cz.cuni.mff.d3s.spl.data.DataSnapshot;
21  import cz.cuni.mff.d3s.spl.data.DataSnapshotBuilder;
22  import cz.cuni.mff.d3s.spl.data.ImmutableBenchmarkRun;
23  import org.junit.Before;
24  import org.junit.Test;
25  
26  import static org.junit.Assert.assertEquals;
27  
28  public class WelchTestWithEnlargedVariancesInterpretationTest {
29  	private static final double EPSILON = 0.00001;
30  	
31  	private static final double[] RUN_1_1 = new double[] {10, 10, 12, 10 };
32  	private static final double[] RUN_1_2 = new double[] {10, 12, 9, 10 };
33  	private static final double[] RUN_2_1 = new double[] {10, 10, 11, 10 };
34  	private static final double[] RUN_2_2 = new double[] {10, 10, 10, 9 };
35  	
36  	private Interpretation interpretation;
37  	
38  	private DataSnapshot SNAPSHOT_1;
39  	private DataSnapshot SNAPSHOT_2;
40  	
41  	@Before
42  	public void prepareSnapshots() {
43  		SNAPSHOT_1 = makeSnapshot(RUN_1_1, RUN_1_2);
44  		SNAPSHOT_2 = makeSnapshot(RUN_2_1, RUN_2_2);
45  	}
46  	
47  	@Before
48  	public void prepareInterpretation() {
49  		interpretation = new WelchTestWithEnlargedVariancesInterpretation();
50  	}
51  	
52  	private DataSnapshot makeSnapshot(double[]... runs) {
53  		DataSnapshotBuilder builder = new DataSnapshotBuilder();
54  		for (double[] samples : runs) {
55  			BenchmarkRun run = new ImmutableBenchmarkRun(samples);
56  			builder.addRun(run);
57  		}
58  		
59  		return builder.create();
60  	}
61  		
62  	
63  	@Test
64  	public void smokeTestForTwoSnapshots() {
65  		ComparisonResult result = interpretation.compare(SNAPSHOT_1, SNAPSHOT_2);
66  		
67  		assertEquals(0.7947194, result.getStatistic(), 0.000001);
68  		assertEquals(ComparisonResult.Relation.GREATER_THAN, result.get(0.3));
69  		assertEquals(-1.281552, result.getCriticalValue(0.1), EPSILON);
70  	}
71  }
